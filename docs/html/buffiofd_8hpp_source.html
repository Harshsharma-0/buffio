724 createSocket(buffioFd *fdCore, struct sockaddr *lsocket, const char *address,
725 int portNumber = 8080,
726 buffioFdFamily family = buffioFdFamily::ipv4,
727 buffioSocketProtocol protocol = buffioSocketProtocol::tcp,
728 bool blocking = false) {
729
730 assert(fdCore != nullptr);
731 assert(lsocket != nullptr);
732 assert(address != nullptr || portNumber > 0);
733 assert(fdCore->fdFamily == buffioFdFamily::none);
734
735 if (portNumber <= 0 || portNumber > 65535)
736 return (int)buffioErrorCode::portnumber;
737
738 int domain = 0;
739 int type = 0;
740 size_t len = 0;
741 sockaddr_storage addr = {0};
742 socklen_t addrLen = 0;
743 char *addressfd = nullptr;
744
745 switch (family) {
746 case buffioFdFamily::ipv4: {
747 domain = AF_INET;
748 type = protocol == buffioSocketProtocol::tcp ? SOCK_STREAM : SOCK_DGRAM;
749
750 sockaddr_in *in4AddrLoc = reinterpret_cast<sockaddr_in *>(&addr);
751 if (inet_pton(domain, address, &in4AddrLoc->sin_addr) != 1)
752 return (int)buffioErrorCode::socketAddress;
753
754 in4AddrLoc->sin_family = domain;
755 in4AddrLoc->sin_port = htons(portNumber);
756 addrLen = sizeof(sockaddr_in);
757 ::memcpy(lsocket, in4AddrLoc, addrLen);
758 } break;
759
760 case buffioFdFamily::ipv6: {
761 domain = AF_INET6;
762 type = protocol == buffioSocketProtocol::tcp ? SOCK_STREAM : SOCK_DGRAM;
763
764 sockaddr_in6 *in6AddrLoc = reinterpret_cast<sockaddr_in6 *>(&addr);
765 if (inet_pton(domain, address, &in6AddrLoc->sin6_addr) != 1)
766 return (int)buffioErrorCode::socketAddress;
767
768 in6AddrLoc->sin6_family = domain;
769 in6AddrLoc->sin6_port = htons(portNumber);
770 addrLen = sizeof(sockaddr_in6);
771
772 ::memcpy(lsocket, in6AddrLoc, addrLen);
773
774 } break;
775 case buffioFdFamily::local: {
776 domain = AF_UNIX;
777 type = protocol == buffioSocketProtocol::tcp ? SOCK_STREAM : SOCK_DGRAM;
778
779 sockaddr_un *unAddrLoc = reinterpret_cast<sockaddr_un *>(&addr);
780 len = ::strlen(address);
781
782#ifdef NDEBUG
783 if (len > sizeof(unAddrLoc->sun_path)) {
784 return (int)buffioErrorCode::socketAddress;
785 };
786
787 try {
788 addressfd = new char[(len + 1)];
789 } catch (std::exception &e) {
790 return (int)buffioErrorCode::makeUnique;
791 };
792#else
793 assert(len > sizeof(unAddrLoc->sun_path));
794 try {
795 addressfd = new char[len + 1];
796 } catch (std::exception &e) {
797 // error error error
798 assert(false);
799 };
800
801#endif // NDEBUG
802
803 ::memcpy(addressfd, address, len);
804 addressfd[len] = '\0';
805 ::memcpy(unAddrLoc->sun_path, addressfd, len + 1);
806
807 addrLen = offsetof(struct sockaddr_un, sun_path) + len + 1;
808 ::memcpy(lsocket, unAddrLoc, addrLen);
809
810 } break;
811 case buffioFdFamily::raw:
812 break;
813 default:
814 return (int)buffioErrorCode::family;
815 break;
816 };
817
818 int socketFd = ::socket(domain, type, 0);
819 if (socketFd < 0)
820 return (int)buffioErrorCode::socket;
821
822 char ip[INET_ADDRSTRLEN];
823 sockaddr_in *tmp = (sockaddr_in *)&addr;
824 inet_ntop(AF_INET, &tmp->sin_addr, ip, sizeof(ip));
825
826 if (::bind(socketFd, reinterpret_cast<sockaddr *>(&addr), addrLen) != 0) {
827 ::close(socketFd);
828
829 return (int)buffioErrorCode::bind;
830 }
831
832 fdCore->fdFamily = family;
833 fdCore->mountSocket(addressfd, socketFd, portNumber);
834 if (blocking == false) {
835 buffioMakeFd::setNonBlocking(socketFd);
836 fdCore->bitSet(BUFFIO_FD_NON_BLOCKING);
837 };
838 return (int)buffioErrorCode::none;
839 };
