94 template <typename P> buffioAwaiter await_transform(P handle) {
95 killChild();
96 std::coroutine_handle<> handleTmp = voidSelf;
97 bool wait = false;
98
99 if constexpr (std::is_same_v<P, buffioTimer *>) {
100 if (handle->then) {
101 auto *promise = getPromise<char>(handle->then);
102 promise->setInstance(clock, broker, fdPool, headerPool);
103 clock->push(handle->duration, handle->then);
104 wait = true;
105 } else {
106 clock->push(handle->duration, voidSelf);
107 };
108 status = buffioRoutineStatus::waitingTimer;
109
110 } else if constexpr (std::is_same_v<P, buffioHeader **>) {
111 *handle = headerPool->pop();
112 wait = true;
113
114 } else if constexpr (std::is_same_v<P, buffioHeader *>) {
115
116 assert(handle != nullptr);
117
118 switch (handle->opCode) {
119 case buffioOpCode::poll:
120 if (broker->pollOp(handle->reqToken.fd, handle->fd,
121 handle->len.mask) == 0) {
122 handle->fd->bitSet(BUFFIO_FD_POLLED);
123 fdPool->pushUse(handle->fd);
124 }
125 break;
126 case buffioOpCode::release:
127 headerPool->push(handle);
128 break;
129 case buffioOpCode::rmPoll:
130 if (*(handle->fd) == BUFFIO_FD_POLLED)
131 broker->pollDel(handle->fd->getFd());
132 break;
133 };
134 wait = true;
135
136 } else if constexpr (std::is_same_v<P, std::coroutine_handle<>>) {
137
138 auto *promise = getPromise<char>(handle);
139 promise->setInstance(clock, broker, fdPool, headerPool);
140 handle_child = handle;
141 status = buffioRoutineStatus::waiting;
142 handleTmp = handle_child;
143
144 } else if constexpr (std::is_same_v<P, buffioFd *>) {
145
146 static_assert(false, "don't support fd not");
147 return {};
148
149 } else if constexpr (std::is_same_v<P, buffioFd **>) {
150 *handle = fdPool->get();
151 wait = true;
152
153 } else {
154 static_assert(false, "we don't support this type");
155 return {};
156 }
